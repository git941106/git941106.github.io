可变参数

*args(arg只是一个规范的命名)接收的是可变参数value,是一个元组，传入的参数会被放进元组里
**kwargs(kwargs只是一个规范的命名)接受的参数是key=value的可变参数,是一个字典，传入的参数以键值对的形式存放到字典里。

def params(*args):
    print(args)# 元组
    print(type(args))

def params_kwargs(**kwargs):
    print(kwargs)
    print(type(kwargs))

if __name__ == "__main__":
    params(1, 2, 3, 4)
    params_kwargs(a=1, b=2)

关于可变参数的解包,解包是在作为参数传递的时候使用的

*args使用的解包是*args，**kwargs的解包是**kwargs

def sum(*args,**kwargs):
    if len(args) != 0:       
        sum = 0
        for i in args:
            sum += i
        return sum
    else:
        a = kwargs.get("a")
        b = kwargs.get("b")
        return a+b


if __name__ == "__main__":
    s = (1, 2)
    su = {"a": 1, "b": 2}
    print(sum(1, 2))
    print(sum(*s)) # 相当于sum(1,2)
    print(sum(**su))  # 相当于sub(a=1,b=2)

装饰器

装饰器包括函数装饰器，类装饰器，装饰器的作用是改变装饰的函数或者类的运行方式，比如在执行函数的时候，先执行输出日志，连接数据库等前置操作，或者执行函数之后，输出函数运行时间等后置操作

无参函数装饰器

from functools import wraps
# 当一个函数作为装饰器时，必须有一个参数，这个参数是使用装饰器的对象，
def login(fun):
    @wraps(func) # 防止使用装饰器的函数的__name__ 和__doc__属性被装饰器取代
    # 如果不加@wraps(func)，order中调用order.__name__ 结果是inner
    def inner():
        print("登录")
        fun()#调用传过来的函数
    return inner
# 当一个函数使用装饰器时，该函数不会被执行，该函数会被传到装饰器中，由装饰器执行，如果装饰器不调用，则该函数
#不会被调用
@login
def order():
    print("下单")


if __name__ == '__main__':
    order()

带参函数装饰器

from functools import wraps
# 当一个函数作为装饰器时，必须有一个参数，这个参数是使用装饰器的对象，
def login(fun):
    @wraps(func) # 防止使用装饰器的函数的__name__ 和__doc__属性被装饰器取代
    def inner(*args,**kwargs):
        '''inner doc'''
        print("登录")
        fun(*args,**kwargs)#调用传过来的函数
    return inner
# 当一个函数使用装饰器时，该函数不会被执行，该函数会被传到装饰器中，
# 由装饰器执行，如果装饰器不调用，则该函数不会被调用
@login
def order(id):
    '''order注释'''
    print(f"下单{id}{order.__name__}{order.__doc__}")


if __name__ == '__main__':
    order()

类装饰器

class T:
    def __init__(self, *args, **kwargs):
        '''
        接收@T(1.5, kw="kw")中的参数，如果是@T()，没有参数，可不写__init__函数
        '''
        self.args = args
        self.kwargs = kwargs

    def __call__(self, func):
        '''接收调用者的函数对象'''
        @wraps(func)
        def inner(*args, **kwargs):
            '''
            接收调用者的函数参数
            '''
            print(f"登录--{self.args}--{args}--{self.kwargs}--{kwargs}")
            func(*args, **kwargs)

        return inner


@T(1.5, kw="kw")
def order(id=1,name="name"):
    '''order doc'''
    print(f"下单{id}--{name}--{order.__doc__}--{order.__name__}")


if __name__ == '__main__':
    order(1,name="name")
#登录--(1.5,)--(1,)--{'kw': 'kw'}--{'name': 'name'}
#下单id:1--name--order doc--order

关键字

global

声明一个变量是全局变量，一般情况是要改变全局变量的值之前，需要声明变量，否则会报错

x = 1
def a():
    print(x)

def b():
    global x # 由于需要修改全局变量，如果注释这句，执行会报错
    x = x + 1
    print(x)

if __name__ == "__main__":
    a()
    b()

yield

yield是返回的作用，代码第一次执行到yield会返回，第二次执行时，会执行yield之后的代码，当函数中存在yield，函数会被当成是一个生成器，生成器可以使用for循环得到每个结果

生成器
a = (x for x in range(1,3))# a得到一个生成器,并不是元组，生成器不能直接打印
b = [x for x in range(1,3)]# b得到的是一个列表
for i in a:
    print(a)
# 也可以使用next(a,None) 第二个参数表示当生成器执行完后，会报StopIteration错误，设置为None，不会报错
def yield_a():
    print("before yield")
    yield 5
    print("after yield")
d = yield_a() # 此时的d就是生成器，不会输出print的结果，可使用for输出
next(d,None)# 获取yield之前代码并执行，包括yield 5,如果不写，返回None
next(d,None)# 获取yield之后代码并执行

加密与解密

base64

经过base64编码后，数据是原来的4/3倍。数据传输的时候，会使用这种编码。编码的原因是为了通用性，数据传输时是二进制，如果不经过编码，不同的设备对原始数据的解读是不一样的。base64编码后，最多包括65个字符(a-z,A-Z,0-9,+,/,=)，其中=是为了编码的可还原加入的补充字符，=只出现再末尾，最多出现2个，原字节数必须是三的倍数，不够就加\x00，编码后加\x00变成=，base64编码后的字符一般都会有+/=

md5

md5是一种不可逆的加密算法，只能通过暴力破解解密,加密后的结果是128bit

import hashlib
class Encryption:
	@staticmethod
    def md5(message: str) -> str:
        md5 = hashlib.md5()
        md5.update(message.encode("utf-8"))
        return md5.hexdigest()

高级加密标准(AES,Advanced Encryption Standard)，是常见的对称加密算法。对称加密是指加密的密匙与解密的密匙相同。

from Crypto.Cipher import AES
class Encryption: 
    @staticmethod
    def aes_encode(message: str, key: bytes = "0102030405060708".encode("utf-8"), mode=AES.MODE_ECB):
        aes = AES.new(key, mode)
        message = message.encode("utf-8")
        while len(message) % 16 != 0:
            message += b'\x00'
        return base64.b64encode(aes.encrypt(message)).decode("utf-8")

    @staticmethod
    def aes_decode(message: str, key: bytes = "0102030405060708".encode("utf-8"), mode=AES.MODE_ECB):
        aes = AES.new(key, mode)
        message = base64.b64decode(message)
        return aes.decrypt(message).decode("utf-8").strip().strip(b'\x00'.decode())
