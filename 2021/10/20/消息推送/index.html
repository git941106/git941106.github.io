def push_message(message, summary="消息摘要"):
    url = "http://wxpusher.zjiecode.com/api/send/message"
    json = {
        "appToken": "",# appToken 从官网获取
        "content": message, # 要发送的消息
        "summary": summary,# 消息摘要
        "contentType": 1,
        "topicIds": [],
        "uids": [
            ""# 发送的 uid从官网获取
        ],
        "url": ""
    }
    requests.post(url,json=json)
